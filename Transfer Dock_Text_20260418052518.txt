        spark.style.width = size + 'px';
        spark.style.height = size + 'px';

        spark.style.background = `hsl(${Math.random() * 60 + 20}, 85%, 65%)`;
        spark.style.boxShadow = '0 0 6px gold';
        document.body.appendChild(spark);
        setTimeout(() => {
            if(spark && spark.remove) spark.remove();
        }, 5000);
    }

            candleElement.innerHTML = '💨✨';
            candleLit = false;

                setTimeout(() => createSparkle(), i * 30);
            }
        } else {
            cakeMessageDiv.innerHTML = '🍰 Lilin sudah padam... saatnya potong kue! 🍰';
        }
    }
